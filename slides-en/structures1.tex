\begin{frame}[fragile]\frametitle{Logical commands}
The document is described in abstract level. Logical command define,
for example,
\begin{itemize}
\item sections (\verb|\section{foo}|),
\item environments (\verb|\begin{frame}|),
\item emphasis (\verb|\emph{observe}|),
\item \dots
\end{itemize}
and the interpretation of these commands is controlled by the style sheet
(.sty) and it depends, of course, also on the media (.pdf document, .pdf
slides or .html page).
\medskip

{\footnotesize Observe that LaTeX was first introduces in 1985 as an extension
for TeX (first released in 1978), and HTML5/CSS3, copying the same idea, is called the
``future of the web'' (as of 2011).}
\end{frame}

\begin{frame}[fragile]\frametitle{What this means?}
Setting the style by hand is not a good idea\footnote{Compare
this with doing <div style=``font-style: italic''> \ldots </div> in an html
document instead of using <emph> and defining that environment properly in
the cascading style sheet (css)!}.
\bigskip

Avoid:
{\small
\begin{verbatim}
\textbf{Backround}\\[3mm]
In the future, one of the most striking 
challenges will be...
\end{verbatim}}

Instead, use:
{\small
\begin{verbatim}
\section{Backround}
In the future, one of the most striking 
challenges will be...
\end{verbatim}}
\end{frame}

\begin{frame}[fragile]\frametitle{What this also means?}
\begin{itemize}
\item LaTeX is a sophisticated, and thus also a bit involved environment.
\item You should be writing scratch files in plain text (or with paper and
pencil -- and then feed the papers into scanner to archive them). 
\item Nevertheless, I use LaTeX to write down the potential ideas still
being immature for publication just to make them clean.
\end{itemize}
\end{frame}


\begin{frame}[fragile]\frametitle{Headings}
The levels of headings are
\begin{itemize}
\item[-1] \verb|\part{}|
\item[0] \verb|\chapter{}|
\item[1] \verb|\section{}|
\item[2] \verb|\subsection{}|
\item[3] \verb|\subsubsection{}|
\item[4] \verb|\paragraph{}|
\item[5] \verb|\subparagraph{}|
\end{itemize}
\end{frame}
 
\begin{frame}[fragile]\frametitle{Headings\dots}
\begin{itemize}
\item The levels  \verb|\part{}| and \verb|\chapter{}| are not used with all
document types. These command are used mostly in books.
\item Ending command (analog to ending tag) is not needed\footnote{This
is in contrast to XML, where the elements always consist of the starting and
ending tag and the content between them. For example,\\ \texttt{<tag>} must
be ended with \texttt{<\textbackslash tag>}.}.
\item Command \verb|\setcounter{secnumdepth}{n}| $n\in {0,\ldots,5}$
defines the extent of explicit numbering (in front of the headings).
\end{itemize} 
\end{frame}

\begin{frame}[fragile]\frametitle{Environments}
The most common environments are
\verb|equation|, \verb|displaymath| and \verb|itemize|. Suppose that
\texttt{end1} and \texttt{env2} are two environments. Then
\begin{verbatim}
\begin{env1}
  ...
  \begin{env2}
  ...
  \end{env2}
\end{env1}
\end{verbatim}
is legal. Observe that the environments need to be nested: \texttt{env1} can
not end before \texttt{env2}.
\end{frame}

\begin{frame}[fragile]\frametitle{Environments: Custom definitions}
Let us define an environment than adds automatically a box at the end of a
proof, and writes the title 'New Proof.' in the beginning: \bigskip
\begin{verbatim}
\newenvironment{newproof}
{\makebox[2cm][l]{\textbf{New Proof.\ }}}
{\hfill $\Box$}
\end{verbatim}
\end{frame}

\begin{frame}[fragile]\frametitle{Environments\dots}
The general form of the environment definition:
\begin{verbatim}
\newenvironment{name}[args]{begdef}{enddef}  
\end{verbatim}
or
\begin{verbatim}
\renewenvironment{name}[args]{begdef}{enddef}
\end{verbatim}
if we wish to re-define an existing environment.
\end{frame}


\begin{frame}[fragile]\frametitle{Environments\dots}
These exists a command for stating lemmas:
\begin{verbatim}
\newtheorem{command name}{theorem name}[counter]
\end{verbatim}
For example, 
\begin{verbatim}
\newtheorem{newlemma}{Lemma}[page]

\begin{newlemma}[The Professors' Remainder 
Theorem]
If we use half of the time left for each 
exercise, we can deal with an infine number 
of exercises. 
\end{newlemma}
\end{verbatim}
\end{frame}

\begin{frame}[fragile]\frametitle{Environments\dots}
\newtheorem{newlemma}{Lemma}[page]
\begin{newlemma}[The Professors' Remainder 
Theorem] 
If we use half of the time left for each 
exercise, we can deal with an infine number 
of exercises.
\end{newlemma}\bigskip

Let us further inspect the command:
\begin{verbatim}
\newtheorem{newlemma}{Lemma}[page]
\end{verbatim}
where the numbering was set to follow the
\texttt{page} counter. Thus, (if this was an article and not a slideshow)
all formulas are numbered as
\begin{center} 
(page,lemma) 
\end{center}
\end{frame}

\begin{frame}[fragile]\frametitle{Commands}
If we need a simple command, e.g. 
\verb|\warning{text}|, one can define
\begin{verbatim}
\newcommand{\warning}[1]{%
\begin{center}
\Large
\shadowbox{
\textbf{#1}}
\end{center}
}
\end{verbatim}
\end{frame}


\begin{frame}[fragile]\frametitle{Commands\dots}
Now, the command
\begin{verbatim}
\warning{orthogonal vectors $\neq$ 
orthonormal vectors}
\end{verbatim}\medskip
produces
\warning{orthogonal vectors $\neq$ orthonormal vectors}
\end{frame}
